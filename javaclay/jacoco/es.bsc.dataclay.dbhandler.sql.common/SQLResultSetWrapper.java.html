	@Override
	public Object getObject(int col) throws SQLException {
		UUID uuid = null;
		String s = rs.getString(col);
		if (s == null) {
			return null;
		}
		try {
			if (s.length() == 36) {
				uuid = UUID.fromString(s);
			}
		} catch (Exception e) {
			return (Object) s;
		}
		return uuid;
	}
